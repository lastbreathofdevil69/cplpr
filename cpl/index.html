1. The 3N + 1 Problem

Problem: Given i and j, find max cycle length for any n in [i,j]. Cycle: n -> n/2 (if even) or n -> 3n+1 (if odd), until n=1.
def cycle_length(n):
    count = 1
    while n != 1:
        if n % 2 == 0:
            n = n // 2
        else:
            n = 3 * n + 1
        count += 1
    return count

i = int(input("Enter starting number (i): "))
j = int(input("Enter ending number (j): "))

lo = min(i, j)
hi = max(i, j)
max_cycle = 0

for n in range(lo, hi + 1):
    current_cycle = cycle_length(n)
    if current_cycle > max_cycle:
        max_cycle = current_cycle

print(f"Result: {i} {j} {max_cycle}")

5. Contest Scoreboard

Problem: Parse submissions (team, problem, time, result). Output ranked scoreboard (More solved -> Less penalty -> Team Number).
n = int(input("Enter number of submissions: "))
teams = {}

print("Enter submissions (team prob time result(C/I)):")
for _ in range(n):
    data = input().split()
    team, prob, time, result = int(data[0]), int(data[1]), int(data[2]), data[3]
    
    if team not in teams:
        teams[team] = {}
    if prob not in teams[team]:
        teams[team][prob] = [0, 0] #[time_solved, wrong_attempts]
    
    if result == 'C' and teams[team][prob][0] == 0:
        teams[team][prob][0] = time
    elif result == 'I' and teams[team][prob][0] == 0:
        teams[team][prob][1] += 1

scoreboard =[]
for team, probs in teams.items():
    solved = 0
    penalty = 0
    for p, stats in probs.items():
        if stats[0] > 0: # If solved
            solved += 1
            penalty += stats[0] + (stats[1] * 20)
    scoreboard.append([team, solved, penalty])

# Sort: Solved (Descending), Penalty (Ascending), Team (Ascending)
scoreboard.sort(key=lambda x: (-x[1], x[2], x[0]))

print("\nFinal Scoreboard (Team Solved Penalty):")
for t, s, p in scoreboard:
    print(t, s, p)

6. Crypt Kicker (Encryption)

Problem: Print the string in a space-separated grid format to encrypt it based on ceiling of the square root (No math library).
def encryption(s):
    # Remove spaces
    s = s.replace(" ", "")
    l = len(s)
    
    # Custom Logic to find Ceil of Square Root
    c = int(l ** 0.5)
    if c * c < l:
        c += 1
        
    result =[]
    # Read characters column-wise
    for i in range(c):
        text = ""
        for j in range(i, l, c):
            text += s[j]
        result.append(text)
        
    return " ".join(result)

# Input
s = input("Enter string: ")
# Output
print("Encrypted:", encryption(s))

7. Queues: A Tale of Two Stacks

Problem: Implement a Queue using two Stacks. Operations: 1=enqueue, 2=dequeue, 3=peek (print front).
stack1 = []
stack2 =[]
q = int(input("Enter number of queries: "))

print("Enter queries (1 x to enqueue, 2 to dequeue, 3 to print):")
for _ in range(q):
    query = list(map(int, input().split()))
    
    # Enqueue operation
    if query[0] == 1:
        stack1.append(query[1])
        
    # Dequeue operation
    elif query[0] == 2:
        if not stack2:
            while stack1:
                stack2.append(stack1.pop())
        if stack2:
            stack2.pop()
            
    # Print front element
    elif query[0] == 3:
        if not stack2:
            while stack1:
                stack2.append(stack1.pop())
        if stack2:
            print(stack2[-1])

8. Castle on the Grid

Problem: Find min moves (one move = full sweep in one direction) from Start to Goal. Uses simple List Queue.
n = int(input("Enter grid size (N): "))
grid =[]
print("Enter the grid row by row (use . for safe, X for block):")
for _ in range(n):
    grid.append(input())
    
sx, sy, gx, gy = map(int, input("Enter Start X, Start Y, Goal X, Goal Y: ").split())

visited = [[False] * n for _ in range(n)]
queue = [[sx, sy, 0]]
visited[sx][sy] = True

found = False
while len(queue) > 0:
    x, y, moves = queue.pop(0) # Standard list pop from front
    
    if x == gx and y == gy:
        print("Minimum moves:", moves)
        found = True
        break
        
    for dx, dy in [[0, 1], [0, -1],[1, 0], [-1, 0]]:
        nx, ny = x + dx, y + dy
        while 0 <= nx < n and 0 <= ny < n and grid[nx][ny] != 'X':
            if not visited[nx][ny]:
                visited[nx][ny] = True
                queue.append([nx, ny, moves + 1])
            nx += dx
            ny += dy

if not found:
    print("Cannot reach the goal.")

10. Wertyu

Problem: Typist shifted one key RIGHT on keyboard. Decode by shifting each character one key LEFT.
keyboard =['`','1','2','3','4','5','6','7','8','9','0','-','=',
            'Q','W','E','R','T','Y','U','I','O','P','[',']','\\',
            'A','S','D','F','G','H','J','K','L',';','\'',
            'Z','X','C','V','B','N','M',',','.','/']
            
text = input("Enter the typed text: ").upper()
result = ""

for ch in text:
    if ch == " ":
        result += " "
    else:
        for i in range(len(keyboard)):
            if keyboard[i] == ch:
                result += keyboard[i - 1]
                break
                
print("Corrected text:", result)

11. Special String Again

Problem: Count "special" substrings. Substrings where all characters are same, or only middle differs.
n = int(input("Enter length of string: "))
s = input("Enter the string: ")
count = 0

for i in range(n):
    count += 1 # Single character is always special
    for j in range(i + 1, n):
        sub = s[i : j + 1]
        
        # Check if all characters are same
        if sub.count(sub[0]) == len(sub):
            count += 1
        # Check if edges same and middle characters all same
        elif sub[0] == sub[-1] and len(sub) > 2:
            middle = sub[1 : -1]
            if middle.count(middle[0]) == len(middle) and middle[0] != sub[0]:
                count += 1
                
print("Total special substrings:", count)

14. Longest Nap

Problem: Given appointments, find the longest free gap between 10:00 and 18:00.
n = int(input("Enter number of appointments: "))
appointments =[]

for i in range(n):
    start, end = input(f"Enter Start and End for appt {i+1} (HH:MM HH:MM): ").split()
    sh, sm = map(int, start.split(':'))
    eh, em = map(int, end.split(':'))
    appointments.append([sh * 60 + sm, eh * 60 + em])

appointments.sort()
current_time = 10 * 60  # 10:00 AM in minutes
day_end = 18 * 60       # 6:00 PM in minutes

best_gap = 0
best_start = current_time

for start, end in appointments:
    if start > current_time:
        gap = start - current_time
        if gap > best_gap:
            best_gap = gap
            best_start = current_time
    current_time = max(current_time, end)

if day_end > current_time:
    gap = day_end - current_time
    if gap > best_gap:
        best_gap = gap
        best_start = current_time

h = best_start // 60
m = best_start % 60
print(f"Longest nap starts at {h}:{m:02d} and lasts for {best_gap} minutes.")

15. Fraudulent Activity Notifications

Problem: Send notification if today's spend >= 2 * median of past d days. Simplest approach for beginners.
n, d = map(int, input("Enter total days and trailing days (n d): ").split())
exp = list(map(int, input("Enter expenditures separated by space: ").split()))

notifications = 0

for i in range(d, n):
    past_days = exp[i - d : i]
    past_days.sort()
    
    if d % 2 == 1:
        median = past_days[d // 2]
    else:
        median = (past_days[d // 2 - 1] + past_days[d // 2]) / 2
        
    if exp[i] >= 2 * median:
        notifications += 1

print("Total notifications sent:", notifications)

20. Combo Meal

Problem: Find minimum cost to buy burgers, fries, and drinks using combo discounts.
prices = input("Enter price of Burger, Fries, Drink, and Combo: ").split()
pb, pf, pd, pc = map(int, prices)

b = int(input("Enter number of Burgers needed: "))
f = int(input("Enter number of Fries needed: "))
d = int(input("Enter number of Drinks needed: "))

min_cost = float('inf')

for combos in range(min(b, f, d) + 1):
    cost = combos * pc
    cost += (b - combos) * pb
    cost += (f - combos) * pf
    cost += (d - combos) * pd
    
    if cost < min_cost:
        min_cost = cost
        
print("Minimum total cost:", min_cost)

21. Steps

Problem: Find minimum steps to cover distance between x and y. Move sizes increment gradually.
t = int(input("Enter number of test cases: "))
for i in range(t):
    x, y = map(int, input("Enter x and y: ").split())
    d = y - x   # Distance
    step = 0    # Number of steps
    move = 0    # Current move length
    total = 0   # Total distance covered
    
    while total < d:
        step = step + 1
        if step % 2 == 1:
            move = move + 1
        total = total + move
        
    print("Minimum steps required:", step)

23. Manasa and Combinatorics

Problem: Compute C(n, r) mod (10^9 + 7).
MOD = 10**9 + 7

# Function to calculate factorial
def factorial(n):
    fact = 1
    for i in range(1, n + 1):
        fact = (fact * i) % MOD
    return fact

# Function to calculate nCr
def nCr(n, r):
    if r > n:
        return 0
    numerator = factorial(n)
    denominator = (factorial(r) * factorial(n - r)) % MOD
    result = (numerator * pow(denominator, MOD - 2, MOD)) % MOD
    return result

# Input
n = int(input("Enter n: "))
r = int(input("Enter r: "))
# Output
print("nCr =", nCr(n, r))

24. Building a List

Problem: Perform left/right operations on a list: append, appendleft, pop, popleft, reverse, print. (No deque library).
n = int(input("Enter number of operations: "))
arr =[]

print("Enter operations (e.g., 'append 5', 'popleft', 'print'):")
for _ in range(n):
    op = input().split()
    
    if op[0] == 'append':
        arr.append(int(op[1]))
    elif op[0] == 'appendleft':
        arr.insert(0, int(op[1]))
    elif op[0] == 'pop':
        if arr:
            print("Popped:", arr.pop())
    elif op[0] == 'popleft':
        if arr:
            print("Popped Left:", arr.pop(0))
    elif op[0] == 'reverse':
        arr.reverse()
    elif op[0] == 'print':
        print("Current List:", arr)
